**Add a helper method `getStationOutputResource(station)`:**
```javascript
getStationOutputResource(station) {
    const world = this.gameState.getWorld();
    const sectorX = Math.floor(station.position.x / world.standardSectorWidth);
    const sectorY = Math.floor(station.position.y / world.standardSectorHeight);
    const sector = world.sectors.get(`${sectorX},${sectorY}`);

    // Backward compatibility: old saves without profiles default to probethium-rich
    // This ensures existing mining stations keep producing probethium (per CONTEXT.md save migration decision)
    const profileType = sector?.resourceProfile?.type || 'probethium-rich';

    const resourceMap = {
        'mineral-rich': 'minerals',
        'data-rich': 'data',
        'artifact-rich': 'artifacts',
        'probethium-rich': 'probethium',
        'balanced': 'mixed'
    };

    return resourceMap[profileType] || 'probethium';
}
```

**Modify production logic in `updateStation()` (around lines 288-292):**

Replace the current hardcoded probethium production block:
```javascript
// CURRENT (remove):
if (continuousProduction > 0) {
    station.totalProduced += continuousProduction;
    this.gameState.mining.totalProbetheum += continuousProduction;
    this.gameState.probethium.current += continuousProduction;
}
```

With profile-based production:
```javascript
if (continuousProduction > 0) {
    const outputResource = this.getStationOutputResource(station);
    station.totalProduced += continuousProduction;

    if (outputResource === 'probethium') {
        this.gameState.mining.totalProbetheum += continuousProduction;
        this.gameState.probethium.current += continuousProduction;
    } else if (outputResource === 'mixed') {
        // Balanced sector: reduced mixed output (0.3x each resource type)
        const perResource = continuousProduction * 0.3;
        const resources = this.gameState.getResources();
        resources.minerals += perResource;
        resources.data += perResource;
        resources.artifacts += perResource;
        this.gameState.updateResources(resources, this.eventBus);
    } else {
        // Specialized sector: full output of specialty resource
        const resources = this.gameState.getResources();
        resources[outputResource] += continuousProduction;
        this.gameState.updateResources(resources, this.eventBus);
    }
}
```
